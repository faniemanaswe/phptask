	 }

		 return $this->name;
	 }
	 
	 public function setName($value){
		 return $this->name = $value;
	 }
	 
	 public function getNumber(){
		 return $this->number;
	 }
	 
	 public function setNumber($value){
		 return $this->number = $value;
     }

        return $this->cost;
    }

}
